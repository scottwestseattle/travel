SELECT
   e.id
 , e.title
FROM 
 entries e
INNER JOIN
 entries_tags
ON
 e.id = entries_tags.entry_id
INNER JOIN 
 tags
ON
 entries_tags.tag_id = tags.id
WHERE
tags.id = 1
